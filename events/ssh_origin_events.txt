### Shellworld
# Set Up
country_event = {
	id = ssh_origin.1
	hide_window = yes

	is_triggered_only = yes

	immediate = {
		save_event_target_as = shellworld_owner
		capital_scope = {
			set_planet_flag = ssh_shellworld_planet_origin
			#generate_start_pops = yes
			# Since they dont spawn with farming districts naturally
			add_deposit = d_fertile_lands
			add_deposit = d_lush_jungle
			add_deposit = d_teeming_reef
			add_deposit = d_submerged_ore_veins
			add_deposit = d_rich_mountain
		}
		country_event = {
			id = ssh_origin.2
			days = 5
		}
	}
}

country_event = {
	id = ssh_origin.2
	title = "ssh_origin.2.name"
	desc = "ssh_origin.2.desc"
	picture = GFX_evt_ssh_shellworld
	show_sound = event_default
	location = from
	trackable = no

	is_triggered_only = yes
	
	option = {
		name = "ssh_origin.2.a"
		add_resource = {
			unity = 150
		}
		add_modifier = {
			modifier = ssh_origin_unity_boost
			days = 360
		}
	}
}

### Gas Giant Shellworld
# Set Up
country_event = {
	id = ssh_origin.3
	hide_window = yes

	is_triggered_only = yes

	immediate = {
		save_event_target_as = shellworld_owner
		species = {
			save_event_target_as = owner_species
		}
		if = {
			limit = { NOT = { has_technology = tech_mine_rare_crystals } }
			give_technology = {
				tech = tech_mine_rare_crystals
			}
		}
		if = {
			limit = { NOT = { has_technology = tech_mine_volatile_motes } }
			give_technology = {
				tech = tech_mine_volatile_motes
			}
		}
		add_resource = {
			unity = 150
		}
		capital_scope = {
			set_planet_flag = ssh_gas_giant_shellworld_planet_origin
		}
	}
}

			limit = { NOT = { has_technology = tech_mine_exotic_gases } }

		add_resource = {
			unity = 150
		}
		capital_scope = {
			set_planet_flag = ssh_edersphere_planet_origin
		}
	}
