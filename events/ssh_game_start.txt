	}
}

system_event = {
	id = ssh_game_start.5
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		every_system_planet = {
			if = {
				limit = { has_modifier = pd_hot_gas_giant }
				add_modifier = {
					modifier = ssh_hot_gas_giant
					days = -1
				}
			}
			if = {
				limit = { has_modifier = pd_cloudless_gas_giant }
				change_pc = pc_gas_giant # Cloudless Gas Giants spawn as Molten Worlds in PD; changing class to actual gas giants so that shellworlds can be constructed around them
				add_modifier = {
					modifier = ssh_cloudless_gas_giant
					days = -1
				}
			}
			if = {
				limit = { is_planet_class = pc_dwarf_gas_giant }
				add_modifier = {
					modifier = ssh_dwarf_gas_giant
					days = -1
				}
			}
		}
	}
}

					ssh_spawn_company_enclave_system = yes
				}
				70 = {
					# No Company
				}
			}
		}
		else_if = {
			limit = { has_global_flag = ssh_company_enclave_guaranteed }
			ssh_spawn_company_enclave_system = yes
		}
		else = {
			# No Company
		}
		# Flavor Stuff
		if = {
			limit = { any_system = { has_star_flag = federation_end } }
			random_system = {
				limit = { has_star_flag = federation_end }
				star = {
					clear_deposits = yes
					add_deposit = d_dark_matter_deposit_10
				}
			}
		}
	}
}
