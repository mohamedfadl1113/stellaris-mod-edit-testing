special_project = {
	key = "SSH_VACDEC_BASEMENT_UNIVERSE_PROBING"
	cost = 80000
	days_to_research = -1
	tech_department = physics_technology
	picture = GFX_evt_ssh_basement_universe

	event_scope = ship_event

	requirements = {
		shipclass_science_ship = 1
		leader = scientist
		skill = 2
	}

	on_success = {
		from.owner = { remove_country_flag = ssh_vacdec_basement_uni_currently_researching_probing_project }
		if = {
			limit = { from.owner = { has_country_flag = probing_black_hole } }
			from.owner = {
				remove_country_flag = probing_black_hole
				country_event = { id = ssh_vacdec.602 }
			}
		}
		else_if = {
			limit = { from.owner = { has_country_flag = probing_neutron_star } }
			from.owner = {
				remove_country_flag = probing_neutron_star
				country_event = { id = ssh_vacdec.603 }
			}
		}
		else_if = {
			limit = { from.owner = { has_country_flag = probing_regular_star } }
			from.owner = {
				remove_country_flag = probing_regular_star
				country_event = { id = ssh_vacdec.604 }
			}
		}
		else_if = {
			limit = { from.owner = { has_country_flag = probing_dwarf_star } }
			from.owner = {
				remove_country_flag = probing_dwarf_star
				country_event = { id = ssh_vacdec.605 }
			}
		}
		else = {
			# Nothing crash
		}
	}

	on_fail = {
		if = {
			limit = { from.owner = { has_country_flag = probing_black_hole } }
			from.owner = {
				remove_country_flag = probing_black_hole
			}
		}
		if = {
			limit = { from.owner = { has_country_flag = probing_neutron_star } }
			from.owner = {
				remove_country_flag = probing_neutron_star
			}
		}
		if = {
			limit = { from.owner = { has_country_flag = probing_regular_star } }
			from.owner = {
				remove_country_flag = probing_regular_star
			}
		}
		if = {
			limit = { from.owner = { has_country_flag = probing_dwarf_star } }
			from.owner = {
				remove_country_flag = probing_dwarf_star
			}
		}
	}
}
