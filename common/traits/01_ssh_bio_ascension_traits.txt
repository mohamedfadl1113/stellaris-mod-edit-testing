#Biologische Waste Reduction (turns waste into xyz)
trait_ssh_waste_bacteria = {
	cost = 4
	initial = no
	randomized = no

	prerequisites = { "ssh_tech_neural_tampering" }

	species_possible_remove = { can_remove_beneficial_genetic_traits = yes }
	species_possible_merge_remove = { always = yes }

	potential_crossbreeding_chance = 1.0

	opposites = { "trait_wasteful" "trait_hive_mind" "trait_ssh_forced_recycling" "trait_ssh_waste_bacteria_lithoid" "trait_invasive" }
	allowed_archetypes = { BIOLOGICAL }

	modifier = {
		planet_pops_consumer_goods_upkeep_mult = -0.50
		planet_pops_food_upkeep_mult = 0.05
	}
	icon = "gfx/interface/icons/traits/trait_ssh_waste_bacteria.dds"
	slave_cost = {
		energy = 1250
	}
}

# Artificially Grown Byproducts
trait_ssh_useful_byproducts = {
	cost = 4
	species_potential_add = { can_do_advanced_gene_modding = yes }
	species_possible_remove = { can_remove_beneficial_genetic_traits = yes }
	species_possible_merge_remove = { always = yes }

	prerequisites = { "ssh_tech_neural_tampering" }

	potential_crossbreeding_chance = 0.5

	initial = no
	randomized = no

	advanced_trait = yes

	opposites = { "trait_ssh_useful_byproducts_lithoid" "trait_invasive" } #species can be both delicious and have useful byproducts
	allowed_archetypes = { BIOLOGICAL }

	tags = { organic positive pop_output food livestock }

	modifier = {
		trader_jobs_bonus_workforce_mult = 0.15
		livestock_jobs_bonus_workforce_mult = 1.5 ### Temporary
	}

	resources = {
		category = planet_pops
		produces = {
			trigger = {
				is_enslaved = yes
				has_slavery_type = { type = slavery_livestock }
			}
			food = 0.015
			minerals = 0.01
			rare_crystals = 0.002
			exotic_gases = 0.002
			volatile_motes = 0.002
		}
	}

	icon = "gfx/interface/icons/traits/trait_ssh_useful_byproducts.dds"
	slave_cost = {
		energy = 1250
	}
	assembly_score = {
		modifier = {
			add = 1.5
			from = { has_farming_designation = yes }
		}
		modifier = {
			add = 0.5
			from = { has_rural_designation = yes }
		}
	}
}
