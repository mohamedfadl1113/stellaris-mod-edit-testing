district_ssh_gas_shellworld_farming = {
	expansion_planner = yes

	base_buildtime = @gas_giant_shellworld_district_buildtime
	is_capped_by_modifier = no

	#min_for_deposits_on_planet = 3
	#max_for_deposits_on_planet = 15

	zone_slots = {
		slot_food
	}

	show_on_uncolonized = {
		exists = from
		from = { is_wilderness_empire = no }
		uses_district_set = gas_shell_world
		NOT = { has_modifier = resort_colony }
	}

	potential = {
		uses_district_set = gas_shell_world
		OR = {
			NOT = { exists = owner }
			NOT = { owner = { is_anglers_empire = yes } }
			AND = {
				owner = { is_anglers_empire = yes }
				NOT = { is_planet_class = pc_ocean }
			}
		}
		NOT = {
			AND = {
				exists = owner
				owner = {
					is_ai = yes
					country_uses_food = no
				}
			}
		}
		OR = {
			NOT = { exists = owner }
			AND = {
				exists = owner
				owner = { is_wilderness_empire = no }
			}
		}
		NOT = { has_modifier = resort_colony }
	}

	conversion_ratio = 1
	convert_to = {
		district_ssh_gas_shellworld_housing
		district_city
		district_nexus
		district_hive
	}

	resources = {
		category = planet_districts_farming
		cost = {
			minerals = 1000
		}
		# Anglers Civic cost effect
		cost = {
			trigger = {
				AND = {
					exists = owner
					owner = { is_anglers_empire = yes }
					planet = { has_climate = wet }
				}
			}
			minerals = -50
		}
		upkeep = {
			energy = 1
		}
	}

	inline_script = {
		script = jobs/farmers_add
		AMOUNT = @gas_giant_shellworld_district_1_job #@gas_giant_shellworld_district_jobs
	}

		potential = {
			exists = owner
			owner = { is_wilderness_empire = no }
		}
		modifier = {
			planet_housing_add = 1500
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_valid_civic = civic_agrarian_idyll
			}
		}
		modifier = {
			planet_housing_add = 400
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_technology = tech_housing_agrarian_idyll
				has_valid_civic = civic_agrarian_idyll
			}
		}
		modifier = {
			planet_housing_add = 200
