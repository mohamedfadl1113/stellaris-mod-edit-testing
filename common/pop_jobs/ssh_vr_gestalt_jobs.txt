		produces = {

			engineering_research = 6
			society_research = 6
		}
	}

	overlord_resources = {
		category = planet_requisitioned_research
		produces = {
			trigger = {
				planet = {
					has_planet_flag = has_science_ministry
				}
			}
			physics_research = 1.5
			engineering_research = 1.5
			society_research = 1.5
		}
	}

		weight = @complex_drone_job_weight

			factor = 0.9
