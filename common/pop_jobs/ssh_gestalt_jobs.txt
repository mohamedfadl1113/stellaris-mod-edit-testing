ssh_gas_collector_drone = {
	category = simple_drone #complex drone? -> Mote harvesting drones are also complex drones
	building_icon = building_gas_extractors

	tags = { strategic_resource exotic_gases }

	possible_pre_triggers = {
		has_owner = yes
		is_enslaved = no
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
	}

	possible_precalc = can_fill_drone_job

	possible = { can_think = yes }

	resources = {
		category = planet_sr_miners
		produces = {
			sr_hydrogen = 5
			exotic_gases = 2
		}
		upkeep = {
			energy = 1
		}
	}

	weight = {
		weight = @simple_drone_job_weight #@complex_drone_job_weight
		mult = value:job_weights_modifier|JOB|ssh_gas_collector_drone|RESOURCE|exotic_gases|
		mult = value:job_weights_modifier|JOB|ssh_gas_collector_drone|RESOURCE|sr_hydrogen|
		modifier = {
			factor = 1.1
			exists = owner
			years_passed < 1
			owner = { has_valid_civic = civic_machine_assimilator }
			has_trait = trait_cybernetic
		}
	}

	promotion = {
		time = @standard_promotion_time
		effect = {
			pop_amount_change_category = {
				CATEGORY = simple_drone #complex_drone
				AMOUNT = local_pop_amount
			}
		}
	}
}
