# Nanites
ssh_nanites = {
	trigger = {
		owner = {
			has_technology = ssh_tech_atomic_manipulation
			OR = {
				NOT = { is_country_type = default }
				has_policy_flag = orbital_bombardment_indiscriminate
				has_policy_flag = orbital_bombardment_armageddon
			}
		}
	}

	default = no

	stop_when_armies_dead = no
	abduct_pops = no

	planet_damage = 1.3
	army_damage = 1.8

	kill_pop_chance = {
		base = 0.9
	}
	min_pops_to_kill_pop = 0

	# root = fleet
	# from = planet
	ai_weight = {
		weight = 5
		modifier = {
			factor = 0.01
			exists = from
			from = {
				owner = { NOT = { is_hostile = root.owner } }
				NAND = {
					controller = { is_crisis_faction = yes }
					planet_garrison_strength > 0
				}
			}
		}
		modifier = {
			factor = 5
			owner = {
				is_materialist = yes
			}
		}
		modifier = {
			factor = 0.1
			owner = {
				OR = {
					is_pacifist = yes
					is_xenophile = yes
					is_spiritualist = yes
				}
			}
		}
	}
}

# Gene Tailored Virus
ssh_virus = {
	trigger = {
		owner = {
			has_technology = ssh_tech_advanced_dna_resequencing #has megastructure bio facility
			OR = {
				NOT = { is_country_type = default }
				has_policy_flag = orbital_bombardment_indiscriminate
				has_policy_flag = orbital_bombardment_armageddon
			}
		}
	}

	default = no

	stop_when_armies_dead = no
	abduct_pops = no

	planet_damage = 0.2
	army_damage = 1.75

	kill_pop_chance = { # Should only target organics
		base = 2.0
		modifier = {
			factor = 0
			OR = {
				pop_group_has_trait = trait_machine_unit
				pop_group_has_trait = trait_mechanical
			}
		}
	}
	min_pops_to_kill_pop = 0

	# root = fleet
	# from = planet
	ai_weight = {
		weight = 10
		modifier = {
			factor = 0.01
			exists = from
			from = {
				owner = { NOT = { is_hostile = root.owner } }
			}
		}
		modifier = {
			factor = 0.01
			owner = {
				OR = {
					is_xenophile = yes
					is_pacifist = yes
					is_spiritualist = yes
				}
			}
		}
	}
}
