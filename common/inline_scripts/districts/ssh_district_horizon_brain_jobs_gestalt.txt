# Triggered Modifier: Add jobs depending on modifiers
# HIVE EMPIRES
triggered_planet_modifier = {
	potential = {
		exists = owner
		owner = {
			is_hive_empire = yes
		}
	}
	modifier = {
		#job_ssh_vr_coordinator_add = 4 #archive drone
		job_ssh_vr_brain_drone_add = $VR_DRONE_AMOUNT$ #4
		job_ssh_vr_patrol_drone_add = $VR_DRONE_AMOUNT$
		job_ssh_vr_warrior_drone_add = $VR_DRONE_AMOUNT$
		job_ssh_vr_infiltration_drone_add = $VR_DRONE_AMOUNT$ #3
	}
}
triggered_planet_modifier = {
	potential = {
		exists = owner
		owner = {
			is_hive_empire = yes
			NOT = { has_valid_civic = civic_hive_memorialist }
		}
	}

		job_ssh_vr_coordinator_add = $VR_DRONE_AMOUNT$
	}
