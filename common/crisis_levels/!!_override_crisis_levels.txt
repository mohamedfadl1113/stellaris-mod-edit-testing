### Additional BIO Brain Behemoth Fury Route ###
################################################

crisis_behemoth_level_5 = {
	# REQUIREMENTS
	allow = {
		is_subject = no
		# Below changed to either regular route or brain route
		OR = {
			custom_tooltip = { # Regular Route
				success_text = crisis_level_5_req_clear_tooltip
				fail_text =	behemoth_crisis_level_5_req_tooltip
				has_country_flag = crisis_special_project_4_complete
			}
			AND = {
				custom_tooltip = { # Brain Route
					success_text = crisis_level_5_req_clear_tooltip
					fail_text =	behemoth_crisis_level_5_megabrain_final_req_tooltip
					has_country_flag = ssh_bio_brain_final
				}
				custom_tooltip = { # Brain Route #2
					success_text = crisis_level_5_req_clear_tooltip
					fail_text =	behemoth_crisis_level_5_megabrain_req_tooltip
					has_country_flag = crisis_special_project_4_megabrain_complete
				}
			}
		}
	}
	required_crisis_currency = 45000

	# REWARDS
	perks = {
		menp_behemoth_mind_meld
		menp_behemoth_primal_roar
		menp_behemoth_ever_hungry
		menp_behemoth_egg_2
	}

	on_unlock = {
		hidden_effect = {
			declare_crisis_effect = yes
			every_playable_country = {
				limit = {
					is_galactic_community_member = no
					NOT = {
						is_same_value = owner
						is_at_war_with = owner
					}
				}
				join_war_on_side = {
					war = event_target:crisis_war
					side = event_target:community_war_leader
				}
			}
			owner = {
				add_event_chain_counter = {
					event_chain = "behemoth_crisis_chain"
					counter = "crisis_level_reached"
					amount = 1
				}
				if = {
					limit = { has_country_flag = behemoth_crisis_megabrain_path }
					country_event = { id = ssh_crisis.3020 days = 3 }
				}
				else = {
					country_event = { id = biocrisis.70 days = 3 } # Regular Path
				}
			}
		}
	}
}
