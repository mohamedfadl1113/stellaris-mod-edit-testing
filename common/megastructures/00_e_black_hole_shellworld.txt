black_hole_shellworld_0 = {
	entity = "construction_platform_entity"
	construction_entity = "construction_platform_entity"
	portrait = "GFX_megastructure_construction_background"
	build_time = 360 #1years
	show_prereqs = yes
	scales_with_planet = no
	place_entity_on_planet_plane = no
	entity_offset = {
		x = -15
		y = -15
	}
	prerequisites = { ssh_tech_black_hole_shellworld }
	potential = {
		has_technology = ssh_tech_black_hole_shellworld
		NOT = {
			has_global_flag = ssh_black_hole_suprashell_deactivated
		}
	}

	possible = {
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}
		custom_tooltip = {
			fail_text = "requires_surveyed_system"
			NOT = {
				any_system_planet = {
					is_surveyed = {
						who = prev.from
						status = no
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = "requires_no_habitable_planets"
			NOT = {
				any_system_planet = {
					colonizable_planet = yes
				}
			}
		}
		custom_tooltip = {
			fail_text = "ssh_requires_black_hole"
			OR = {
				is_star_class = sc_black_hole
				is_star_class = sc_ssh_binary_1
				is_star_class = sc_ssh_binary_2
				# Hypo Stars (Binary Black Hole Systems)
				is_star_class = sc_hypo_binary_1
				is_star_class = sc_hypo_binary_4
			}
		}
		custom_tooltip = {
			fail_text = "requires_no_existing_megastructure"
			has_no_non_gate_megastructure = yes
		}
	}

	resources = {
		category = megastructures
		cost = {
			unity = 5000
		}
		cost = {
			trigger = { country_uses_bio_ships = no }
			alloys = 5000
		}
		cost = {
			trigger = { country_uses_bio_ships = yes }
			alloys = 5000
			mult = 0.5
		}
		cost = {
			trigger = { country_uses_bio_ships = yes }
			food = 5000
			mult = @halved_alloy_to_food_cost_ratio
		}
		upkeep = {
			energy = 10
		}
	}

	placement_rules = {
		planet_possible = {
			custom_tooltip = {
				fail_text = "ssh_requires_black_hole"
				is_planet_class = pc_black_hole
			}
			custom_tooltip = {
				fail_text = "requires_no_existing_megastructure"
				NOR = {
					has_planet_flag = has_megastructure
					has_planet_flag = megastructure
				}
			}
			custom_tooltip = {
				fail_text = "requires_no_anomaly"
				NOT = { has_anomaly = yes }
			}
		}
	}

	on_build_complete = {
		if = {
			limit = { exists = fromfrom.planet }
			fromfrom.planet = {
				set_planet_flag = has_megastructure
				save_event_target_as = black_hole_shellworld_planet
			}
		}
		from = {
			set_country_flag = ssh_is_constructing_black_hole_shellworld
			country_event = { id = ssh_cn.76 }
		}
	}

	ai_weight = {
		weight = 15
		# Do we need it? (Other colonies not maxed out yet?)
		modifier = {
			factor = 0.1
			from = {
				any_system_within_border = {
					any_system_colony = {
						num_free_districts = {
							type = any
							value > 3
						}
					}
				}
			}
		}
		# Do we need the resources?
		modifier = {
			factor = 2.0
			from = {
				OR = {
					has_monthly_income = {
						resource = sr_dark_matter
						value < 10
					}
					has_monthly_income = {
						resource = sr_cultrobium
						value < 20
					}
				}
			}
		}
		# Do we have enough planets already?
		# Are we constructing one already? (in case of unlimited mega build capacity)
		modifier = {
			factor = 0
			from = { has_country_flag = ssh_is_constructing_black_hole_shellworld }
		}
		# Do we have enough resources?
		modifier = {
			factor = 0.1
			from = {
				AND = {
					resource_stockpile_compare = {
						resource = energy
						value < 25000
					}
					resource_stockpile_compare = {
						resource = minerals
						value < 35000
					}
					resource_stockpile_compare = {
						resource = sr_cultrobium
						value < 28000
					}
					resource_stockpile_compare = {
						resource = alloys
						value < 45000
					}
					resource_stockpile_compare = {
						resource = exotic_gases
						value < 5100
					}
				}
			}
		}
		# Upkeep Affordable?
		modifier = {
			factor = 0.5
			from = {
				has_monthly_income = {
					resource = energy
					value < 100
				}
			}
		}
		# Better habitable megastructures researched already + possibility to build them?
		# >>>>>>>>>>> TODO: Add giga's habitable megas to the list (alderson, birch world)

		# Good System?
		modifier = {
			factor = 0.1
			starbase = { NOT = { has_starbase_size >= starbase_starfortress } }
		}
		modifier = {
			factor = 0.1
			any_neighbor_system = {
				exists = owner
				NOT = {
					owner = { is_same_value = from }
				}
			}
		}
		# Personality?
		modifier = {
			factor = 1.2
			from = {
				OR = {
					has_ai_personality = xenophobic_isolationists
					has_ai_personality = fanatic_purifiers
					has_ai_personality = erudite_explorers
					has_ai_personality = imperial_origin_overlord_ai
	
					has_authority = auth_corporate
					has_authority = auth_hive_mind
					has_authority = auth_machine_intelligence
					has_civic = civic_technocracy
				}
			}
		}
	}
}

					ssh_is_any_gas_giant = yes # keep for gas giant compressor -> star -> agri districts
					ssh_is_any_regular_star = yes # keep for agri districts
				}
			}
			remove_planet = yes
		}
		from = { country_event = { id = ssh_cn.77 } }
	}
}

black_hole_shellworld_6 = {
	entity = "ssh_black_hole_shellworld_6_entity"
	construction_entity = "ssh_black_hole_shellworld_6_entity"
	portrait = "GFX_megastructure_construction_background"
	build_time = 7200 #20years
	scales_with_planet = yes
	place_entity_on_planet_plane = yes
	prerequisites = { ssh_tech_black_hole_shellworld }
	potential = { has_technology = ssh_tech_black_hole_shellworld }

	resources = {
		category = megastructures
		cost = {
			minerals = 30000
			exotic_gases = 5000
			energy = 20000
		}
		upkeep = {
			energy = 50
		}
	}

	upgrade_from = {
		black_hole_shellworld_5
	}

	on_build_complete = {
		set_star_flag = ssh_black_hole_suprashell_system # Used for cosmogenesis ending
		from = {
			set_country_flag = ssh_has_built_or_repaired_tier_e_mega
			set_country_flag = has_built_or_repaired_megastructure # Galwonders
			set_country_flag = has_built_or_repaired_gigastructure # Unlocks Gigastructures' AP
			remove_country_flag = ssh_is_constructing_black_hole_shellworld
			country_event = { id = ssh_cn.82 }
		}
		fromfrom.planet = {
			set_planet_flag = has_megastructure
		}
		if = {
			limit = { exists = fromfrom.planet }
			spawn_planet = {
				class = pc_black_hole_shellworld
				location = fromfrom.planet
				orbit_angle_offset = 0
				init_effect = {
					set_name = "Black Hole Suprashell"
					set_planet_size = 40
					set_planet_entity = {
						entity = "ssh_black_hole_shellworld_planet_01_entity"
					}
					set_surveyed = {
						surveyed = yes
						surveyor = fromfrom
					}
					set_all_comms_surveyed = yes
					clear_blockers = yes
					save_event_target_as = black_hole_shellworld
					trigger_megastructure_icon = yes
					set_planet_flag = megastructure
					set_planet_flag = black_hole_shellworld
					add_deposit = d_black_hole_suprashell_deposit
				}
			}
			event_target:black_hole_shellworld = {
				add_modifier = {
					modifier = "ssh_black_hole_shellworld_planet_modifier"
					days = -1
				}
				if = {
					limit = { any_system_planet = { ssh_is_any_regular_star = yes } }
					set_planet_flag = ssh_black_hole_shell_has_star #allows for agri districts
				}
				random_list = {
					25 = { add_deposit = d_black_hole_suprashell_motes_deposit }
					25 = { add_deposit = d_black_hole_suprashell_zro_deposit }
					50 = {
						# No special deposit
					}
				}
				if = {
					limit = {
						solar_system = {
							OR = {
								has_star_flag = great_wound
								has_star_flag = ssh_warped_skies_system
							}
						}
					}
					add_deposit = d_black_hole_suprashell_great_wound_deposit
				}
			}
		}
		remove_megastructure = fromfrom
	}
}
