captive_star_0 = {
	entity = "construction_platform_entity"
	construction_entity = "construction_platform_entity"
	portrait = "GFX_megastructure_construction_background"
	build_time = 720 #2 years
	show_prereqs = yes
	scales_with_planet = no
	place_entity_on_planet_plane = no
	entity_offset = {
		x = -15
		y = -15
	}
	prerequisites = { ssh_tech_captive_star }
	potential = {
		has_technology = ssh_tech_captive_star
		NOT = {
			has_global_flag = ssh_captive_star_deactivated
		}
	}

	possible = {
		exists = starbase
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}
		custom_tooltip = {
			fail_text = "requires_surveyed_system"
			NOT = {
				any_system_planet = {
					is_surveyed = {
						who = prev.from
						status = no
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = "requires_no_binary_trinary"
			NOR = {
				is_binary_star = yes
				is_trinary_star = yes
			}
		}
		custom_tooltip = {
			fail_text = "ssh_requires_no_colony"
			NOT = {
				any_system_planet = {
					is_colony = yes
				}
			}
		}
		custom_tooltip = {
			fail_text = "ssh_requires_no_black_hole"
			NOT = { is_star_class = sc_black_hole }
		}
		custom_tooltip = {
			fail_text = "ssh_requires_no_special_star"
			NOT = { ssh_is_special_star = yes }
		}
		custom_tooltip = {
			fail_text = "ssh_requires_no_neutron"
			NOR = { 
				is_star_class = sc_neutron_star
				is_star_class = sc_pulsar
				is_star_class = sc_ssh_quark_star
				# Hypo Stars
				is_star_class = sc_strange_star
			}
		}
		custom_tooltip = {
			fail_text = "ssh_requires_no_black_dwarf"
			NOT = {
				is_star_class = sc_ssh_black_dwarf
			}
		}
		custom_tooltip = {
			fail_text = "requires_no_existing_megastructure"
			has_no_non_gate_megastructure = yes
		}
	}

	resources = {
		category = megastructures
		cost = {
			influence = 50
		}
		cost = {
			trigger = { country_uses_bio_ships = no }
			alloys = 2500
		}
		cost = {
			trigger = { country_uses_bio_ships = yes }
			alloys = 2500
			mult = 0.5
		}
		cost = {
			trigger = { country_uses_bio_ships = yes }
			food = 2500
			mult = @halved_alloy_to_food_cost_ratio
		}
		upkeep = {
			energy = 10
		}
	}

	placement_rules = {
		planet_possible = {
			custom_tooltip = {
				fail_text = "must_build_around_star"
				is_star = yes
				NOR = {
					is_planet_class = pc_black_dwarf_star
					is_planet_class = pc_black_hole
					is_planet_class = pc_pulsar
					is_planet_class = pc_neutron_star
					is_planet_class = pc_quark_star
					ssh_is_special_star = yes #Gigas Galactic Core etc
				}
			}
			custom_tooltip = {
				fail_text = "requires_no_existing_megastructure"
				NOT = {
					has_planet_flag = has_megastructure
					#has_planet_flag = megastructure
				}
			}
			custom_tooltip = {
				fail_text = "requires_no_anomaly"
				NOT = { has_anomaly = yes }
			}
		}
	}

	on_build_complete = {
		if = {
			limit = { exists = fromfrom.planet }
			fromfrom.planet = {
				set_planet_flag = has_megastructure
				set_planet_flag = captive_star_planet
				save_event_target_as = stellar_harvester_planet
				orbital_station = {
					dismantle = yes
				}
			}
		}
		from = {
			set_country_flag = ssh_is_constructing_captive_star
			country_event = { id = ssh_cn.150 }
		}
	}

	ai_weight = {
		weight = 15
		# Any OP rival?
		modifier = {
			factor = 5.0
			from = {
				num_rivals > 0
				any_rival_country = {
					relative_power = {
						who = prev
						category = fleet
						value = pathetic #overwhelming
					}
				}
			}
		}
		# Are we constructing one already? (in case of unlimited mega build capacity)
		modifier = {
			factor = 0
			from = { has_country_flag = ssh_is_constructing_captive_star }
		}
		# Do we have enough resources?
		modifier = {
			factor = 0.1
			from = {
				resource_stockpile_compare = {
					resource = alloys
					value < 10000
				}
			}
		}
		# Upkeep Affordable?
		modifier = {
			factor = 0.1
			from = {
				has_monthly_income = {
					resource = alloys
					value < 20
				}
			}
		}

		modifier = {
			factor = 0.1
			any_neighbor_system = {
				exists = owner
				NOT = {
					owner = { is_same_value = from }
				}
			}
		}
		# Personality?
		modifier = {
			factor = 1.5
			from = {
				OR = {
					is_militarist = yes
					is_gestalt = yes
				}
			}
		}
		modifier = {
			factor = 0.5
			from = {
				is_pacifist = yes
			}
		}
	}
}
