stellar_harvester_0 = {
	entity = "construction_platform_entity"
	construction_entity = "construction_platform_entity"
	portrait = "GFX_megastructure_construction_background"
	build_time = 360 #1year
	show_prereqs = yes
	scales_with_planet = no
	place_entity_on_planet_plane = no
	entity_offset = {
		x = -15
		y = -15
	}
	prerequisites = { ssh_tech_stellar_harvester }
	potential = {
		has_technology = ssh_tech_stellar_harvester
		NOT = {
			has_global_flag = ssh_stellar_harvester_deactivated
		}
	}

	possible = {
		exists = starbase
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}
		custom_tooltip = {
			fail_text = "requires_surveyed_system"
			NOT = {
				any_system_planet = {
					is_surveyed = {
						who = prev.from
						status = no
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = "ssh_requires_no_colony"
			NOT = {
				any_system_planet = {
					is_colony = yes
				}
			}
		}
		custom_tooltip = {
			fail_text = "ssh_requires_no_black_hole"
			NOT = { is_star_class = sc_black_hole }
		}
		custom_tooltip = {
			fail_text = "ssh_requires_no_special_star"
			NOT = { ssh_is_special_star = yes }
		}
		custom_tooltip = {
			fail_text = "ssh_requires_no_neutron"
			NOR = { 
				is_star_class = sc_neutron_star
				is_star_class = sc_pulsar
				is_star_class = sc_ssh_quark_star
				# Hypo Stars
				is_star_class = sc_strange_star
			}
		}
		custom_tooltip = {
			fail_text = "ssh_requires_no_black_dwarf"
			NOT = {
				is_star_class = sc_ssh_black_dwarf
			}
		}
		custom_tooltip = {
			fail_text = "requires_no_existing_megastructure"
			has_no_non_gate_megastructure = yes
		}
	}

	resources = {
		category = megastructures
		cost = {
			influence = 50
		}
		cost = {
			trigger = { country_uses_bio_ships = no }
			alloys = 2500
		}
		cost = {
			trigger = { country_uses_bio_ships = yes }
			alloys = 2500
			mult = 0.5
		}
		cost = {
			trigger = { country_uses_bio_ships = yes }
			food = 2500
			mult = @halved_alloy_to_food_cost_ratio
		}
		upkeep = {
			energy = 10
		}
	}

	placement_rules = {
		planet_possible = {
			custom_tooltip = {
				fail_text = "must_build_around_star"
				is_star = yes
				NOR = {
					is_planet_class = pc_black_dwarf_star
					is_planet_class = pc_black_hole
					is_planet_class = pc_pulsar
					is_planet_class = pc_neutron_star
					is_planet_class = pc_quark_star
					ssh_is_special_star = yes #Gigas Galactic Core etc
				}
			}
			custom_tooltip = {
				fail_text = "requires_no_existing_megastructure"
				NOT = {
					has_planet_flag = has_megastructure
					#has_planet_flag = megastructure
				}
			}
			custom_tooltip = {
				fail_text = "requires_no_anomaly"
				NOT = { has_anomaly = yes }
			}
		}
	}

	on_build_complete = {
		if = {
			limit = { exists = fromfrom.planet }
			fromfrom.planet = {
				set_planet_flag = has_megastructure
				set_planet_flag = active_stellar_harvester_planet
				save_event_target_as = stellar_harvester_planet
				orbital_station = {
					dismantle = yes
				}
			}
		}
		from = {
			set_country_flag = ssh_is_constructing_stellar_harvester
			country_event = { id = ssh_cn.84 }
		}
	}

	ai_weight = {
		weight = 15
		# Do we need the resources?
		modifier = {
			factor = 2.0
			from = {
				has_monthly_income = {
					resource = energy
					value < 100
				}
			}
		}
		# Are we constructing one already? (in case of unlimited mega build capacity)
		modifier = {
			factor = 0
			from = { has_country_flag = ssh_is_constructing_stellar_harvester }
		}
		# Do we have enough resources?
		modifier = {
			factor = 0.1
			from = {
				resource_stockpile_compare = {
					resource = alloys
					value < 35000
				}
			}
		}
		# Upkeep Affordable?
		modifier = {
			factor = 0.1
			from = {
				has_monthly_income = {
					resource = alloys
					value < 30
				}
			}
		}

		modifier = {
			factor = 0.1
			any_neighbor_system = {
				exists = owner
				NOT = {
					owner = { is_same_value = from }
				}
			}
		}
		# Personality?
		modifier = {
			factor = 1.15
			from = {
				OR = {
					has_ai_personality = erudite_explorers
					has_authority = auth_corporate
					has_authority = auth_hive_mind
					has_authority = auth_machine_intelligence
					has_ethic = ethic_fanatic_materialist
					has_civic = civic_technocracy
					is_anthropocentric = yes # E&C
				}
			}
		}
	}
}

stellar_harvester_dismantle = {
	entity = "ssh_stellar_harvester_active_entity"
	construction_entity = "ssh_stellar_harvester_active_entity"
	portrait = "GFX_megastructure_construction_background"
	build_time = 360
	
	upgrade_from = {
		stellar_harvester_active
	}

	possible = {
		from = {
			has_technology = ssh_tech_stellar_harvester
		}
	}

	ai_weight = {
		factor = 0
	}

	on_build_complete = {
		fromfrom = {
			remove_megastructure = this
		}
		fromfrom.planet = {
			remove_planet_flag = has_megastructure
			remove_planet_flag = active_stellar_harvester_planet
		}
		# Set according star class (if not trinary/binary
		if = {
			limit = {
				NOR = {
					is_binary_star = yes
					is_trinary_star = yes
				}
			}
			fromfrom.planet = {
				# only the classes the star can be transformed into by the harvester matter
				if = {
					limit = { is_planet_class = pc_g_star }
					solar_system = { set_star_class = sc_g }
				}
				else_if = {
					limit = { is_planet_class = pc_m_star }
					solar_system = { set_star_class = sc_m }
				}
				else_if = {
					limit = { is_planet_class = pc_blue_dwarf_star }
					solar_system = { set_star_class = sc_ssh_blue_dwarf }
				}
				else_if = {
					limit = { is_planet_class = pc_white_dwarf_star }
					solar_system = { set_star_class = sc_ssh_white_dwarf }
				}
				else_if = {
					limit = { is_planet_class = pc_black_dwarf_star }
					solar_system = { set_star_class = sc_ssh_black_dwarf }
				}
			}
		}
	}
}
