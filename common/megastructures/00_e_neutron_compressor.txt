neutron_compressor_0 = {
	entity = "construction_platform_entity"
	construction_entity = "construction_platform_entity"
	portrait = "GFX_megastructure_construction_background"
	build_time = 1800 #5years
	show_prereqs = yes
	scales_with_planet = no
	place_entity_on_planet_plane = no
	entity_offset = {
		x = -15
		y = -15
	}
	prerequisites = { ssh_tech_neutronium_compression }
	potential = {
		has_technology = ssh_tech_neutronium_compression
		NOT = {
			has_global_flag = ssh_neutronium_compressor_deactivated
		}
	}

	possible = {
		exists = starbase
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}
		custom_tooltip = {
			fail_text = "requires_surveyed_system"
			NOT = {
				any_system_planet = {
					is_surveyed = {
						who = prev.from
						status = no
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = "requires_no_habitable_planets"
			NOT = {
				any_system_planet = {
					colonizable_planet = yes
				}
			}
		}
		# ---> binary neutron star systems exist
		#custom_tooltip = {
		#	fail_text = "ssh_requires_neutron"
		#	OR = {
		#		is_star_class = sc_pulsar
		#		is_star_class = sc_neutron_star
		#	}
		#}
	}

	resources = {
		category = megastructures
		cost = {
			influence = 50
		}
		cost = {
			trigger = { country_uses_bio_ships = no }
			alloys = 2500
		}
		cost = {
			trigger = { country_uses_bio_ships = yes }
			alloys = 2500
			mult = 0.5
		}
		cost = {
			trigger = { country_uses_bio_ships = yes }
			food = 2500
			mult = @halved_alloy_to_food_cost_ratio
		}
		upkeep = {
			energy = 10
		}
	}

	placement_rules = {
		planet_possible = {
			custom_tooltip = {
				fail_text = "must_build_around_star"
				is_star = yes
			}
			custom_tooltip = {
				fail_text = "ssh_requires_neutron"
				OR = {
					is_planet_class = pc_neutron_star
					is_planet_class = pc_pulsar
				}
			}
			custom_tooltip = {
				fail_text = "requires_no_existing_megastructure"
				NOT = {
					has_planet_flag = has_megastructure
					#has_planet_flag = megastructure
				}
			}
			custom_tooltip = {
				fail_text = "requires_no_anomaly"
				NOT = { has_anomaly = yes }
			}
		}
	}

	on_build_complete = {
		if = {
			limit = { exists = fromfrom.planet }
			fromfrom.planet = {
				set_planet_flag = has_megastructure
				set_planet_flag = active_neutron_compressor_planet
				save_event_target_as = neutron_compressor_planet
				orbital_station = {
					dismantle = yes
				}
			}
		}
		from = {
			set_country_flag = ssh_is_constructing_neutron_compressor
			country_event = { id = ssh_cn.98 }
		}
	}

	ai_weight = {
		weight = 10
		# Do we need the resources?
		modifier = {
			factor = 2.0
			from = {
				has_monthly_income = {
					resource = sr_cultrobium
					value < 50
				}
			}
		}
		# Are we constructing one already? (in case of unlimited mega build capacity)
		modifier = {
			factor = 0
			from = { has_country_flag = ssh_is_constructing_neutron_compressor }
		}
		# Do we have enough resources?
		modifier = {
			factor = 0.1
			from = {
				resource_stockpile_compare = {
					resource = alloys
					value < 25000
				}
				#resource_stockpile_compare = {
				#	resource = sr_cultrobium
				#	value < 2500
				#}
				resource_stockpile_compare = {
					resource = minerals
					value < 5000
				}
				resource_stockpile_compare = {
					resource = energy
					value < 10000
				}
			}
		}
		# Upkeep Affordable?
		modifier = {
			factor = 0.1
			from = {
				AND = {
					has_monthly_income = {
						resource = minerals
						value < 1000
					}
					has_monthly_income = {
						resource = energy
						value < 2100
					}
				}
			}
		}

		modifier = {
			factor = 0.1
			any_neighbor_system = {
				exists = owner
				NOT = {
					owner = { is_same_value = from }
				}
			}
		}
		# Personality?
		modifier = {
			factor = 1.5
			from = {
				OR = {
					has_ai_personality = erudite_explorers
	
					has_authority = auth_corporate
					has_authority = auth_hive_mind
					has_authority = auth_machine_intelligence
					has_ethic = ethic_fanatic_materialist
					has_civic = civic_technocracy
				}
			}
		}
	}
}
